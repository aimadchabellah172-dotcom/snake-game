    squares.append(i ** 2)

# دالة بسيطة
def greet(name):
    return f"أهلاً {name}! 👋"

# معاملات افتراضية
def power(base, exp=2):
    return base ** exp

print(power(5))      # 25 (5²)
print(power(2, 10))  # 1024 (2¹⁰)

# *args — عدد غير محدد من المعاملات
def total(*numbers):
    return sum(numbers)

print(total(1, 2, 3))           # 6
print(total(10, 20, 30, 40))    # 100

# **kwargs — معاملات مسماة
def create_card(**info):
    for key, val in info.items():
        print(f"  {key}: {val}")

create_card(name="أحمد", age=20, city="القاهرة")

ما هي البرمجة الكائنية؟

طريقة لتنظيم الكود في "كائنات" تمثل أشياء حقيقية. مثلاً: السيارة لها صفات (لون، سرعة) وأفعال (تشغيل، توقف)

🏗️ إنشاء Class

oop_car.py 
class Car:
    """كلاس يمثل سيارة"""
    
    # متغير مشترك بين جميع السيارات
    wheels = 4
    
    def __init__(self, brand, color, speed=0):
        self.brand = brand    # ماركة السيارة
        self.color = color    # اللون
        self.speed = speed    # السرعة الحالية
    
    def accelerate(self, amount):
        self.speed += amount
        print(f"🚗 {self.brand}: السرعة الآن {self.speed} كم/س")
    
    def brake(self):
        self.speed = 0
        print(f"🛑 {self.brand}: توقفت")
    
    def __str__(self):
        return f"سيارة {self.brand} ({self.color})"


# إنشاء كائنات
car1 = Car("Toyota", "أبيض")
car2 = Car("BMW", "أسود")

car1.accelerate(60)   # Toyota: السرعة الآن 60 كم/س
car1.accelerate(40)   # Toyota: السرعة الآن 100 كم/س
car1.brake()           # Toyota: توقفت

print(car2)            # سيارة BMW (أسود)

inheritance.py 
class Animal:
    def __init__(self, name):
        self.name = name
    
    def speak(self):
        return "..."
    
    def __str__(self):
        return f"{self.name} يقول: {self.speak()}"


class Dog(Animal):    # يرث من Animal
    def speak(self):
        return "هاو هاو! 🐕"

class Cat(Animal):
    def speak(self):
        return "ميياو! 🐱"

class Duck(Animal):
    def speak(self):
        return "كواك! 🦆"


animals = [Dog("ريكس"), Cat("كيتي"), Duck("دونالد")]
for animal in animals:
    print(animal)

import tkinter as tk

class Calculator:
    def __init__(self, root):
        self.root = root
        self.root.title("🔢 الآلة الحاسبة")
        self.root.geometry("300x400")
        self.root.configure(bg="#1e1e2e")
        self.root.resizable(False, False)
        
        self.expression = ""
        self.create_widgets()
    
    def create_widgets(self):
        # شاشة العرض
        self.display = tk.Entry(
            self.root, font=("Arial", 24),
            bg="#2d2d3f", fg="#f7c948",
            bd=0, justify="right", insertbackground="white"
        )
        self.display.pack(fill="x", padx=10, pady=10)
        
        # أزرار الحاسبة
        buttons = [
            ['C', '⌫', '%', '/'],
            ['7', '8', '9', '*'],
            ['4', '5', '6', '-'],
            ['1', '2', '3', '+'],
            ['0', '.', '**', '=']
        ]
        
        for row_data in buttons:
            frame = tk.Frame(self.root, bg="#1e1e2e")
            frame.pack(fill="x", padx=5, pady=2)
            for btn_text in row_data:
                color = "#58a6ff" if btn_text in ['+','-','*','/','='] \
                       else "#ff7b72" if btn_text == 'C' \
                       else "#2d2d3f"
                tk.Button(
                    frame, text=btn_text,
                    font=("Arial", 16, "bold"),
                    bg=color, fg="white", bd=0,
                    width=4, height=2,
                    command=lambda t=btn_text: self.click(t)
                ).pack(side="left", padx=2, expand=True, fill="x")
    
    def click(self, char):
        if char == '=':
            try:
                result = eval(self.expression)
                self.display.delete(0, tk.END)
                self.display.insert(0, str(result))
                self.expression = str(result)
            except:
                self.display.delete(0, tk.END)
                self.display.insert(0, "خطأ")
                self.expression = ""
        elif char == 'C':
            self.expression = ""
            self.display.delete(0, tk.END)
        elif char == '⌫':
            self.expression = self.expression[:-1]
            self.display.delete(0, tk.END)
            self.display.insert(0, self.expression)
        else:
            self.expression += char
            self.display.delete(0, tk.END)
            self.display.insert(0, self.expression)


root = tk.Tk()
app = Calculator(root)
root.mainloop()

import tkinter as tk
from tkinter import filedialog, messagebox

def new_file():
    text_area.delete("1.0", tk.END)
    root.title("مفكرة جديدة")

def open_file():
    file = filedialog.askopenfilename(
        filetypes=[("ملفات نصية", "*.txt")]
    )
    if file:
        with open(file, encoding="utf-8") as f:
            text_area.delete("1.0", tk.END)
            text_area.insert("1.0", f.read())
        root.title(file)

def save_file():
    file = filedialog.asksaveasfilename(
        defaultextension=".txt",
        filetypes=[("ملفات نصية", "*.txt")]
    )
    if file:
        with open(file, "w", encoding="utf-8") as f:
            f.write(text_area.get("1.0", tk.END))
        messagebox.showinfo("تم", "✅ تم حفظ الملف!")

import random

def hangman():
    words = ["python", "programming", "computer",
             "algorithm", "variable", "function"]
    
    word = random.choice(words)
    guessed = set()
    attempts = 6
    
    print("🎮 لعبة الكلمة المخفية!")
    print(f"الكلمة: {len(word)} حرف\n")
    
    while attempts > 0:
        display = [c if c in guessed else "_" for c in word]
        print(" ".join(display))
        
        if "_" not in display:
            print(f"\n🎉 مبروك! الكلمة: {word}")
            return
        
        print(f"المحاولات المتبقية: {attempts}")
        print(f"الحروف المجرّبة: {', '.join(sorted(guessed))}")
        
        letter = input("اختر حرفاً: ").lower()
        
        if letter in guessed:
            print("⚠️ جرّبت هذا الحرف من قبل!")
        elif letter in word:
            guessed.add(letter)
            print("✅ إجابة صحيحة!")
        else:
            guessed.add(letter)
            attempts -= 1
            print(f"❌ خطأ! المحاولات: {attempts}")
        print()
    
    print(f"💀 انتهت المحاولات! الكلمة كانت: {word}")

hangman()
